**循环添加数据**
SQL> create or replace procedure print_test is
  2  num number:=1;
  3  begin
  4     loop
  5     insert into cdxtest values(num,num);
  6     exit when num = 10;
  7     num := num+1;
  8     end loop;
  9     commit;
 10  end;
